  list_intersect("${SWIFT_EXPERIMENTAL_FOUNDATION_OVERLAY_SDKS_FROM_REPO}" "${SWIFT_SDKS}" TARGET_SDKS_FOR_XCODE_OVERLAY)

  #TODO this should be a parameter that CMake receives
  set(FOUNDATION_OVERLAY_SOURCE_DIR "${CMAKE_SOURCE_DIR}/../swift-corelibs-foundation/Darwin")

  add_overlay_xcode_project(Foundation
    SOURCE_DIR ${FOUNDATION_OVERLAY_SOURCE_DIR}
    BUILD_TARGET Foundation-swiftoverlay
    ADDITIONAL_XCODEBUILD_ARGUMENTS ${ADDITIONAL_OVERLAY_XCODEBUILD_ARGS}
    DEPENDS CoreFoundationOverlay
    TARGET_SDKS ${TARGET_SDKS_FOR_XCODE_OVERLAY})
  add_overlay_targets(Foundation TARGET_SDKS ${TARGET_SDKS_FOR_XCODE_OVERLAY})
endif()
